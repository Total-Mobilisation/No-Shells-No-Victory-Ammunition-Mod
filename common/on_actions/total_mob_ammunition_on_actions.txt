#Distributes industrial stats to all relevant nations.
#Used by functions to convert into actual production.
#Seven key stats including:
#Mils, Dockyards and Civilian factories
#Steel factories
#Total factor productivity, factorylabour and weeklyhours.

on_actions = {
	on_startup = {
		effect = {
			every_country = {
				#Scripted effects
				total_mob_daily_xp_change_calculator = yes
				total_mob_army_composition_effect = yes
				total_mob_ammunition_consumption_effect = yes
				total_mob_ammunition_calculation_effects = yes
				
				#Add ammunition
				set_variable = { ammunition_to_add = max_daily_ammo_use }
				multiply_variable = { ammunition_to_add = 15 }

				#Starting stockpile
				add_equipment_to_stockpile = {
					type = ammunition_equipment
    				amount = var:ammunition_to_add
				}
			}
		}
	}
	on_daily = {
		effect = {
			add_dynamic_modifier = { modifier = ammo_adjustment_modifier }
			#Modifiers
			total_mob_daily_xp_change_calculator = yes
			total_mob_army_composition_effect = yes
			total_mob_ammunition_consumption_effect = yes
			total_mob_ammunition_calculation_effects = yes
			total_mob_ai_ammunition_production_effect = yes
		}
	}
}
