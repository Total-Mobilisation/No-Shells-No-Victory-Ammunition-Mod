#Distributes industrial stats to all relevant nations.
#Used by functions to convert into actual production.
#Seven key stats including:
#Mils, Dockyards and Civilian factories
#Steel factories
#Total factor productivity, factorylabour and weeklyhours.

on_actions = {
	on_startup = {
		effect = {
			every_country = {
				#Set stats
				if = {
					limit = {TAG=FRA}
					#1936 Stats
					set_variable = { 1936_infantry_equipment_stockpile = 220000 }
					set_variable = { 1936_artillery_equipment_stockpile = 2000 }
					set_variable = { 1936_support_equipment_stockpile = 945 }
					set_variable = { 1936_motorized_equipment_stockpile = 25000 }
					set_variable = { 1936_light_tank_stockpile = 1400 }
					#1939 Stats
					set_variable = { 1939_infantry_equipment_stockpile = 220000 }
					set_variable = { 1939_artillery_equipment_stockpile = 2000 }
					set_variable = { 1939_support_equipment_stockpile = 945 }
					set_variable = { 1939_motorized_equipment_stockpile = 25000 }
					set_variable = { 1939_light_tank_stockpile = 1400 }
				}
				distribute_starting_stockpiles = yes
			}
		}
	}
}
